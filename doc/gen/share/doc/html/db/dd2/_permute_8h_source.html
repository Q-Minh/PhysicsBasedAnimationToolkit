42 static_assert(

46 using PtrDiffType = decltype(n);
47 for (PtrDiffType i = 0; i < n; ++i)
48 {
49 auto pi = *(pb + i);
50 if (pi < 0)
51 continue;

62 *(pb + xi) -= n;
63 }

66 *(pb + i) += n;
67 }
68}
