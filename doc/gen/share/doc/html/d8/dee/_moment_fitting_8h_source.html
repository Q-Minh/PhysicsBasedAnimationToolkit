132template <class TDerivedP, class TDerivedB>
133VectorX MomentFittedWeights(
134 Eigen::MatrixBase<TDerivedP> const& P,
135 Eigen::DenseBase<TDerivedB> const& b,
136 Index maxIterations = 10,
137 Scalar precision = std::numeric_limits<Scalar>::epsilon())
138{
139 using MatrixType = TDerivedP;
140 Eigen::NNLS<MatrixType> nnls{};
141 nnls.setMaxIterations(maxIterations);
142 nnls.setTolerance(precision);
143 nnls.compute(P.derived());
144 auto w = nnls.solve(b);
145 if (nnls.info() != Eigen::ComputationInfo::Success)
146 {
147 std::string what = "Moment fitting's non-negative least-squares failed with error: ";
148 if (nnls.info() == Eigen::ComputationInfo::InvalidInput)
149 what += "InvalidInput";
150 if (nnls.info() == Eigen::ComputationInfo::NoConvergence)
151 what += "NoConvergence";
152 if (nnls.info() == Eigen::ComputationInfo::NumericalIssue)
153 what += "NumericalIssue";
154 throw std::invalid_argument(what);
155 }
156 return w;
157}

170template <CPolynomialBasis Polynomial, class TDerivedXg, class TDerivedWg>
171Vector<Polynomial::kSize> Integrate(
172 Polynomial const& P,
173 Eigen::MatrixBase<TDerivedXg> const& Xg,
174 Eigen::DenseBase<TDerivedWg> const& wg)
175{
176 Vector<Polynomial::kSize> b{};
177 b.setZero();
178 for (auto g = 0; g < wg.size(); ++g)
179 b += wg(g) * P.eval(Xg.col(g));
180 return b;
181}

245template <
246 auto Order,
247 class TDerivedS1,
248 class TDerivedXi1,
249 class TDerivedS2,
250 class TDerivedXi2,
251 class TDerivedWg2>
252std::pair<VectorX, VectorX> TransferQuadrature(
253 Eigen::DenseBase<TDerivedS1> const& S1,
254 Eigen::MatrixBase<TDerivedXi1> const& Xi1,
255 Eigen::DenseBase<TDerivedS2> const& S2,
256 Eigen::MatrixBase<TDerivedXi2> const& Xi2,
257 Eigen::DenseBase<TDerivedWg2> const& wi2,
258 Index nSimplices = -1,
259 bool bEvaluateError = false,
260 Index maxIterations = 10,
261 Scalar precision = std::numeric_limits<Scalar>::epsilon())
262{
263 // Compute adjacency graph from simplices s to their quadrature points Xi
264 using common::ArgSort;
265 using common::Counts;
266 using common::CumSum;
267 using common::ToEigen;
268 if (nSimplices < 0)
269 nSimplices = std::max(
270 *std::max_element(S1.begin(), S1.end()),
271 *std::max_element(S2.begin(), S2.end())) +
272 1;
273 IndexVectorX S1P = CumSum(Counts(S1.begin(), S1.end(), nSimplices));
274 IndexVectorX S2P = CumSum(Counts(S2.begin(), S2.end(), nSimplices));
275 IndexVectorX S1N = ArgSort<Index>(S1.size(), [&](auto si, auto sj) { return S1(si) < S1(sj); });
276 IndexVectorX S2N = ArgSort<Index>(S2.size(), [&](auto si, auto sj) { return S2(si) < S2(sj); });
277 // Find weights wg1 that fit the given quadrature rule Xi2, wi2 on simplices S2
278 auto fSolveWeights = [maxIterations,
279 precision,
280 bEvaluateError](auto const& Xg1, auto const& Xg2, auto const& wg2) {
281 if (Xg1.rows() == 1)
282 {
283 OrthonormalPolynomialBasis<1, Order> P{};
284 auto w = TransferQuadrature(P, Xg1, Xg2, wg2, maxIterations, precision);
285 Scalar error(0);
286 if (bEvaluateError)
287 {
288 auto b1 = math::Integrate(P, Xg1, w);
289 auto b2 = math::Integrate(P, Xg2, wg2);
290 error = (b1 - b2).squaredNorm();
291 }
292 return std::make_pair(w, error);
293 }
294 if (Xg1.rows() == 2)
295 {
296 OrthonormalPolynomialBasis<2, Order> P{};
297 auto w = TransferQuadrature(P, Xg1, Xg2, wg2, maxIterations, precision);
298 Scalar error(0);
299 if (bEvaluateError)
300 {
301 auto b1 = math::Integrate(P, Xg1, w);
302 auto b2 = math::Integrate(P, Xg2, wg2);
303 error = (b1 - b2).squaredNorm();
304 }
305 return std::make_pair(w, error);
306 }
307 if (Xg1.rows() == 3)
308 {
309 OrthonormalPolynomialBasis<3, Order> P{};
310 auto w = TransferQuadrature(P, Xg1, Xg2, wg2, maxIterations, precision);
311 Scalar error(0);
312 if (bEvaluateError)
313 {
314 auto b1 = math::Integrate(P, Xg1, w);
315 auto b2 = math::Integrate(P, Xg2, wg2);
316 error = (b1 - b2).squaredNorm();
317 }
318 return std::make_pair(w, error);
319 }
320 throw std::invalid_argument(
321 "Expected quadrature points in reference simplex space of dimensions (i.e. rows) 1,2 "
322 "or 3.");
323 };
324 // Solve moment fitting on each simplex
325 VectorX error = VectorX::Zero(nSimplices);
326 VectorX wi1 = VectorX::Zero(Xi1.cols());
327 tbb::parallel_for(Index(0), nSimplices, [&](Index s) {
328 auto S1begin = S1P(s);
329 auto S1end = S1P(s + 1);
330 if (S1end > S1begin)
331 {
332 auto s1inds = S1N(Eigen::seq(S1begin, S1end - 1));
333 MatrixX Xg1 = Xi1(Eigen::placeholders::all, s1inds);
334 auto S2begin = S2P(s);
335 auto S2end = S2P(s + 1);
336 auto s2inds = S2N(Eigen::seq(S2begin, S2end - 1));
337 MatrixX Xg2 = Xi2(Eigen::placeholders::all, s2inds);
338 VectorX wg2 = wi2(s2inds);
339 auto [wg1, err] = fSolveWeights(Xg1, Xg2, wg2);
340 wi1(s1inds) = wg1;
341 error(s) = err;
342 }
343 });
344 return {wi1, error};
345}

374template <
375 int Order,
376 class TDerivedS1,

381std::tuple<MatrixX /*P*/, MatrixX /*B*/, IndexVectorX /*prefix into columns of P*/>
382ReferenceMomentFittingSystems(
383 Eigen::DenseBase<TDerivedS1> const& S1,
384 Eigen::MatrixBase<TDerivedX1> const& X1,
385 Eigen::DenseBase<TDerivedS2> const& S2,
386 Eigen::MatrixBase<TDerivedX2> const& X2,
387 Eigen::DenseBase<TDerivedW2> const& w2,
388 Index nSimplices = Index(-1))
389{
390 // Compute adjacency graph from simplices s to their quadrature points Xi
391 using common::ArgSort;
392 using common::Counts;
393 using common::CumSum;
394 using common::ToEigen;
395 if (nSimplices < 0)
396 nSimplices = std::max(
397 *std::max_element(S1.begin(), S1.end()),
398 *std::max_element(S2.begin(), S2.end())) +
399 1;
400 IndexVectorX S1P = CumSum(Counts<Index>(S1.begin(), S1.end(), nSimplices));
401 IndexVectorX S2P = CumSum(Counts<Index>(S2.begin(), S2.end(), nSimplices));
402 IndexVectorX S1N = ArgSort<Index>(S1.size(), [&](auto si, auto sj) { return S1(si) < S1(sj); });
403 IndexVectorX S2N = ArgSort<Index>(S2.size(), [&](auto si, auto sj) { return S2(si) < S2(sj); });
404 // Assemble moment fitting matrices and their rhs
405 auto fPolyRows = [](MatrixX const& Xg) {
406 if (Xg.rows() == 1)
407 return OrthonormalPolynomialBasis<1, Order>::kSize;
408 if (Xg.rows() == 2)
409 return OrthonormalPolynomialBasis<2, Order>::kSize;
410 if (Xg.rows() == 3)
411 return OrthonormalPolynomialBasis<3, Order>::kSize;
412 throw std::invalid_argument(
413 "Expected quadrature points in reference simplex space of dimensions (i.e. rows) 1,2 "
414 "or 3.");
415 };
416 auto fAssembleSystem =
417 [](auto const& Xg1, auto const& Xg2, auto const& wg2) -> std::pair<MatrixX, VectorX> {
418 if (Xg1.rows() == 1)
419 {
420 OrthonormalPolynomialBasis<1, Order> P{};
421 auto M = ReferenceMomentFittingMatrix(P, Xg1);
422 auto b = Integrate(P, Xg2, wg2);
423 return {M, b};
424 }
425 if (Xg1.rows() == 2)
426 {
427 OrthonormalPolynomialBasis<2, Order> P{};
428 auto M = ReferenceMomentFittingMatrix(P, Xg1);
429 auto b = Integrate(P, Xg2, wg2);
430 return {M, b};
431 }
432 if (Xg1.rows() == 3)
433 {
434 OrthonormalPolynomialBasis<3, Order> P{};
435 auto M = ReferenceMomentFittingMatrix(P, Xg1);
436 auto b = Integrate(P, Xg2, wg2);
437 return {M, b};
438 }
439 throw std::invalid_argument(
440 "Expected quadrature points in reference simplex space of dimensions (i.e. rows) 1,2 "
441 "or 3.");
442 };
443 auto nrows = fPolyRows(X1);
444
445 // Count actual number of systems and their dimensions
446 Index nSystems{0};
447 for (Index s = 0; s < nSimplices; ++s)
448 if (S1P(s + 1) > S1P(s))
449 ++nSystems;
450 IndexVectorX nQuads(nSystems);
451 nQuads.setZero();
452 for (Index s = 0, sy = 0; s < nSimplices; ++s)
453 {
454 if (S1P(s + 1) > S1P(s))
455 {
456 nQuads(sy++) = S1P(s + 1) - S1P(s);
457 }
458 }
459 IndexVectorX const prefix = CumSum(nQuads);
460
461 MatrixX P(nrows, prefix(Eigen::placeholders::last));
462 MatrixX B(nrows, nSystems);
463 B.setZero();
464 for (Index s = 0, sy = 0; s < nSimplices; ++s)
465 {
466 auto S1begin = S1P(s);
467 auto S1end = S1P(s + 1);
468 if (S1end > S1begin)
469 {
470 auto s1inds = S1N(Eigen::seq(S1begin, S1end - 1));
471 MatrixX Xg1 = X1(Eigen::placeholders::all, s1inds);
472 auto S2begin = S2P(s);
473 auto S2end = S2P(s + 1);
474 auto s2inds = S2N(Eigen::seq(S2begin, S2end - 1));
475 MatrixX Xg2 = X2(Eigen::placeholders::all, s2inds);
476 VectorX wg2 = w2(s2inds);
477 auto [Ps, bs] = fAssembleSystem(Xg1, Xg2, wg2);
478 P.block(0, prefix(sy), nrows, nQuads(sy)) = Ps;
479 B.col(sy) = bs;
480 ++sy;
481 }
482 }
483 return std::make_tuple(P, B, prefix);
484}

499template <class TDerivedM, class TDerivedP>
500CSRMatrix BlockDiagonalReferenceMomentFittingSystem(
501 Eigen::MatrixBase<TDerivedM> const& M,
502 Eigen::DenseBase<TDerivedP> const& P)
503{
504 auto const nblocks = P.size() - 1;
505 auto const nblockrows = M.rows();
506 auto const nrows = nblockrows * nblocks;
507 auto const ncols = P(Eigen::placeholders::last);
508 CSRMatrix GM(nrows, ncols);
509 IndexVectorX reserves(nrows);
510 for (auto b = 0; b < nblocks; ++b)
511 {
512 auto begin = P(b);
513 auto end = P(b + 1);
514 auto const nblockcols = end - begin;
515 auto const offset = b * nblockrows;
516 for (auto i = 0; i < nblockrows; ++i)
517 reserves(offset + i) = nblockcols;
518 }
519 GM.reserve(reserves);
520 for (auto b = 0; b < nblocks; ++b)
521 {
522 auto begin = P(b);
523 auto end = P(b + 1);
524 auto const nblockcols = end - begin;
525 auto Mb = M.block(0, begin, nblockrows, nblockcols);
526 auto const roffset = b * nblockrows;
527 auto const coffset = begin;
528 for (auto i = 0; i < nblockrows; ++i)
529 {
530 for (auto j = 0; j < nblockcols; ++j)
531 {
532 GM.insert(roffset + i, coffset + j) = Mb(i, j);
533 }
534 }
535 }
536 return GM;
537}
