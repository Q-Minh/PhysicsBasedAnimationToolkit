81template <auto kDims>
82template <auto kSimplexVerts>
83inline void Aabb<kDims>::Construct(
84 common::Buffer<GpuScalar, kDims> const& V,
85 common::Buffer<GpuIndex, kSimplexVerts> const& S,
86 GpuIndex begin)
87{
88 PBAT_PROFILE_NAMED_CUDA_HOST_SCOPE_START(ctx, "pbat.gpu.impl.geometry.Aabb.Construct");
89 using namespace pbat::math::linalg;
90 auto const nSimplices = static_cast<GpuIndex>(S.Size());
91 if (Size() < nSimplices)
92 Resize(nSimplices);
93 auto const end = std::min(begin + nSimplices, Size());
94 thrust::for_each(
95 thrust::device,
96 thrust::counting_iterator(begin),
97 thrust::counting_iterator(end),
98 [begin, b = b.Raw(), e = e.Raw(), V = V.Raw(), S = S.Raw()] PBAT_DEVICE(GpuIndex i) {
99 auto s = i - begin;
100 auto inds = mini::FromBuffers<kSimplexVerts, 1>(S, s);
101 auto P = mini::FromBuffers(V, inds.Transpose());
102 pbat::common::ForRange<0, kDims>([i, b, e, &P] PBAT_DEVICE<auto d>() {
103 b[d][i] = Min(P.Row(d));
104 e[d][i] = Max(P.Row(d));
105 });
106 });
107 PBAT_PROFILE_CUDA_HOST_SCOPE_END(ctx);
108}
