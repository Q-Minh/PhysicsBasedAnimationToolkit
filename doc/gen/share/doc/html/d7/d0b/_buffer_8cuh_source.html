110template <class T, int D>
111inline std::vector<T> Buffer<T, D>::Get(std::size_t count) const
112{
113 PBAT_PROFILE_NAMED_CUDA_HOST_SCOPE_START(ctx, "pbat.gpu.impl.common.Buffer.CopyToCpu");
114 if (count > Size())
115 {
116 std::string const what = "Requested " + std::to_string(count) +
117 " buffer elements, but buffer has size " + std::to_string(Size());
118 throw std::invalid_argument(what);
119 }
120 std::vector<T> buffer(count * D);
121 for (auto d = 0; d < D; ++d)
122 {
123 thrust::copy(mBuffers[d].begin(), mBuffers[d].begin() + count, buffer.begin() + d * count);
124 }
125 PBAT_PROFILE_CUDA_HOST_SCOPE_END(ctx);
126 return buffer;
127}
