160template <class TDerivedP>
161inline IndexVectorX TriangleAabbHierarchy3D::PrimitivesContainingPoints(
162 Eigen::MatrixBase<TDerivedP> const& P,
163 bool bParallelize) const
164{
165 PBAT_PROFILE_NAMED_SCOPE("pbat.geometry.TriangleAabbHierarchy3D.PrimitivesContainingPoints");
166 using math::linalg::mini::FromEigen;
167 IndexVectorX p(P.cols());
168 p.setConstant(-1);
169 auto const FindContainingPrimitive = [&](Index i) {
170 std::vector<Index> const intersectingPrimitives = this->PrimitivesIntersecting(
171 [&](BoundingVolumeType const& bv) -> bool { return bv.contains(P.col(i)); },
172 [&](PrimitiveType const& T) -> bool {
173 auto const VT = V(Eigen::placeholders::all, T);
174 Scalar const d = DistanceQueries::PointTriangle(
175 FromEigen(P.col(i).template head<kDims>()),
176 FromEigen(VT.col(0).head<kDims>()),
177 FromEigen(VT.col(1).head<kDims>()),
178 FromEigen(VT.col(2).head<kDims>()));
179 auto constexpr eps = 1e-15;
180 return d < eps;
181 });
182 if (not intersectingPrimitives.empty())
183 {
184 p(i) = intersectingPrimitives.front();
185 }
186 };
187 if (bParallelize)
188 {
189 tbb::parallel_for(Index{0}, Index{P.cols()}, FindContainingPrimitive);
190 }
191 else
192 {
193 for (auto i = 0; i < P.cols(); ++i)
194 FindContainingPrimitive(i);
195 }
196 return p;
197}

199template <class TDerivedP, class FCull>
200inline auto TriangleAabbHierarchy3D::NearestPrimitivesToPoints(
201 Eigen::MatrixBase<TDerivedP> const& P,
202 FCull fCull,
203 bool bParallelize) const -> std::pair<IndexVectorX, VectorX>
204{
205 PBAT_PROFILE_NAMED_SCOPE("pbat.geometry.TriangleAabbHierarchy3D.NearestPrimitivesToPoints");
206 using math::linalg::mini::FromEigen;
207 IndexVectorX p(P.cols());
208 p.setConstant(-1);
209 VectorX d(P.cols());
210 d.setConstant(std::numeric_limits<Scalar>::max());
211 auto const FindNearestPrimitive = [&](Index i) {
212 std::size_t constexpr K{1};
213 auto const [nearestPrimitives, distances] = this->NearestPrimitivesTo(
214 [&](BoundingVolumeType const& bv) -> Scalar {
215 return bv.squaredExteriorDistance(P.col(i));
216 },
217 [&](PrimitiveType const& T) -> Scalar {
218 if (fCull(i, T))
219 {
220 return std::numeric_limits<Scalar>::max();
221 }
222 auto const VT = V(Eigen::placeholders::all, T);
223 return DistanceQueries::PointTriangle(
224 FromEigen(P.col(i).template head<kDims>()),
225 FromEigen(VT.col(0).head<kDims>()),
226 FromEigen(VT.col(1).head<kDims>()),
227 FromEigen(VT.col(2).head<kDims>()));
228 },
229 K);
230 p(i) = nearestPrimitives.front();
231 d(i) = distances.front();
232 };
233 if (bParallelize)
234 {
235 tbb::parallel_for(Index{0}, Index{P.cols()}, FindNearestPrimitive);
236 }
237 else
238 {
239 for (auto i = 0; i < P.cols(); ++i)
240 FindNearestPrimitive(i);
241 }
242 return {p, d};
243}
