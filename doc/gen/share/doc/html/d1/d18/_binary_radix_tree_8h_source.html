211inline void
212BinaryRadixTree<TIndex>::Construct(Eigen::DenseBase<TDerived> const& codes, bool bStoreParent)
213{
214 using CodeType = typename TDerived::Scalar;
215 static_assert(
216 std::is_integral_v<CodeType> and std::is_unsigned_v<CodeType> and
217 not std::is_same_v<CodeType, bool>,
218 "Codes must be of unsigned integral type");
219 auto constexpr nBits = sizeof(CodeType) * 8;
220 static_assert(nBits <= 64, "CodeType must have at most 64 bits");
221
222 std::uint64_t constexpr msb = 0b1ULL << (nBits - 1);
223 TIndex const nLeaves = codes.size();
224 TIndex const nInternal = nLeaves - 1;
225 mChild.resize(2, nInternal);
226 mParent.resize(bStoreParent * (nLeaves + nInternal));
227
228 struct Node
229 {
230 TIndex begin;
231 TIndex end;
232 };
233 auto const fCommonPrefixLength = [](CodeType ci, CodeType cj) {
234 return std::countl_zero(ci ^ cj);
235 };
236 pbat::common::Stack<Node, 64> stack{};
237 stack.Push({0, nLeaves - 1});
238 // Top-down construction over internal nodes
239 while (not stack.IsEmpty())
240 {
241 Node const node = stack.Pop();
242 // Compute range [first, last] of codes covered by the node.
243 // If the node is a left child, its range is reversed (i.e. end -> begin).
244 // If the node is a right child, its range is not reversed (i.e. begin -> end).
245 bool const bReversed = node.begin > node.end;
246 auto const first = (not bReversed) * node.begin + bReversed * node.end;
247 auto const last = (not bReversed) * node.end + bReversed * node.begin;
248 // Find the split position
249 TIndex split = first;
250 auto const cfirst = codes(first);
251 auto const clast = codes(last);
252 if (cfirst == clast)
253 {
254 split += (last - first + 1) / 2;
255 }
256 else
257 {
258 auto const mask = msb >> fCommonPrefixLength(cfirst, clast);
259 auto const begin = codes.begin() + first;
260 auto const end = codes.begin() + last + 1;
261 auto const upper = std::upper_bound(begin, end, cfirst, [&](CodeType ci, CodeType cj) {
262 return (mask & ci) < (mask & cj);
263 });
264 split += std::distance(begin, upper);
265 }
266 // The left and right child ranges are split as [first, split-1] and [split,last],
267 // respectively.
268 TIndex lc = split - 1;
269 TIndex rc = split;
270 // Ranges of size 1 indicate leaf nodes. We offset leaf indices by the number of internal
271 // nodes.
272 bool const bIsLeftLeaf = (lc == first);
273 bool const bIsRightLeaf = (rc == last);
274 lc += bIsLeftLeaf * nInternal;
275 rc += bIsRightLeaf * nInternal;
276 // Set parent-child relationships
277 mChild(0, node.begin) = lc;
278 mChild(1, node.begin) = rc;
279 // Only recurse into internal nodes
280 if (not bIsLeftLeaf)
281 stack.Push({lc, first});
282 if (not bIsRightLeaf)
283 stack.Push({rc, last});
284 }
285 // Store parent relationships if requested
286 if (bStoreParent)
287 {
288 mParent(0) = 0;
289 for (auto i = 0; i < nInternal; ++i)
290 {
291 mParent(Left(i)) = i;
292 mParent(Right(i)) = i;
293 }
294 }
295}
