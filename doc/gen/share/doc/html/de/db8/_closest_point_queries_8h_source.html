144template <mini::CMatrix TMatrixX, mini::CMatrix TMatrixP, mini::CMatrix TMatrixN>
145PBAT_HOST_DEVICE auto PointOnPlane(TMatrixX const& X, TMatrixP const& P, TMatrixN const& n)
146 -> mini::SVector<typename TMatrixX::ScalarType, TMatrixX::kRows>
147{
148 using namespace std;
149 using ScalarType = typename TMatrixX::ScalarType;
150#ifndef NDEBUG
151 bool const bIsNormalUnit = abs(SquaredNorm(n) - ScalarType(1)) <= ScalarType(1e-15);
152 assert(bIsNormalUnit);
153#endif
157 ScalarType const t = Dot(n, X - P);
158 mini::SVector<ScalarType, TMatrixX::kRows> Xplane = X - t * n;
159 return Xplane;
160}

162template <mini::CMatrix TMatrixX, mini::CMatrix TMatrixP, mini::CMatrix TMatrixQ>
163PBAT_HOST_DEVICE auto PointOnLineSegment(TMatrixX const& X, TMatrixP const& P, TMatrixQ const& Q)
164 -> mini::SVector<typename TMatrixX::ScalarType, TMatrixX::kRows>
165{
166 using ScalarType = typename TMatrixX::ScalarType;
167 using namespace std;
171 mini::SVector<ScalarType, TMatrixX::kRows> const PQ = Q - P;
172 // Project X onto PQ, computing parameterized position R(t) = P + t*(Q ï¿½ P)
173 ScalarType t = Dot(X - P, PQ) / SquaredNorm(PQ);
174 // If outside segment, clamp t (and therefore d) to the closest endpoint
175 t = min(max(t, ScalarType(0)), ScalarType(1));
176 // Compute projected position from the clamped t
177 auto const Xpq = P + t * PQ;
178 return Xpq;
179}

181template <mini::CMatrix TMatrixX, mini::CMatrix TMatrixL, mini::CMatrix TMatrixU>
182PBAT_HOST_DEVICE auto
183PointOnAxisAlignedBoundingBox(TMatrixX const& P, TMatrixL const& L, TMatrixU const& U)
184 -> mini::SVector<typename TMatrixX::ScalarType, TMatrixX::kRows>
185{
186 using namespace std;
190 mini::SVector<typename TMatrixX::ScalarType, TMatrixX::kRows> X = P;
191 pbat::common::ForRange<0, TMatrixX::kRows>(
192 [&]<auto i>() { X(i) = min(max(X(i), L(i)), U(i)); });
193 return X;
194}

196template <
197 mini::CMatrix TMatrixP,
198 mini::CMatrix TMatrixA,
199 mini::CMatrix TMatrixB,
200 mini::CMatrix TMatrixC>
201PBAT_HOST_DEVICE auto
202UvwPointInTriangle(TMatrixP const& P, TMatrixA const& A, TMatrixB const& B, TMatrixC const& C)
203 -> mini::SVector<typename TMatrixP::ScalarType, 3>
204{
205 using ScalarType = typename TMatrixP::ScalarType;
209
210 // Check if P in vertex region outside A
211 auto constexpr kRows = TMatrixP::kRows;
212 mini::SVector<ScalarType, kRows> const AB = B - A;
213 mini::SVector<ScalarType, kRows> const AC = C - A;
214 mini::SVector<ScalarType, kRows> const AP = P - A;
215 ScalarType const d1 = Dot(AB, AP);
216 ScalarType const d2 = Dot(AC, AP);
217 if (d1 <= ScalarType(0) and d2 <= ScalarType(0))
218 {
219 return mini::Unit<ScalarType, 3>(0); // barycentric coordinates (1,0,0)
220 }
221
222 // Check if P in vertex region outside B
223 mini::SVector<ScalarType, kRows> const BP = P - B;
224 ScalarType const d3 = Dot(AB, BP);
225 ScalarType const d4 = Dot(AC, BP);
226 if (d3 >= ScalarType(0) and d4 <= d3)
227 {
228 return mini::Unit<ScalarType, 3>(1); // barycentric coordinates (0,1,0)
229 }
230
231 // Check if P in edge region of AB, if so return projection of P onto AB
232 ScalarType const vc = d1 * d4 - d3 * d2;
233 if (vc <= ScalarType(0) and d1 >= ScalarType(0) and d3 <= ScalarType(0))
234 {
235 ScalarType const v = d1 / (d1 - d3);
236 return mini::SVector<ScalarType, 3>{
237 ScalarType(1) - v,
238 v,
239 ScalarType(0)}; // barycentric coordinates (1-v,v,0)
240 }
241
242 // Check if P in vertex region outside C
243 mini::SVector<ScalarType, kRows> const CP = P - C;
244 ScalarType const d5 = Dot(AB, CP);
245 ScalarType const d6 = Dot(AC, CP);
246 if (d6 >= ScalarType(0) and d5 <= d6)
247 {
248 return mini::Unit<ScalarType, 3>(2); // barycentric coordinates (0,0,1)
249 }
250
251 // Check if P in edge region of AC, if so return projection of P onto AC
252 ScalarType const vb = d5 * d2 - d1 * d6;
253 if (vb <= ScalarType(0) and d2 >= ScalarType(0) and d6 <= ScalarType(0))
254 {
255 ScalarType const w = d2 / (d2 - d6);
256 return mini::SVector<ScalarType, 3>{
257 ScalarType(1) - w,
258 ScalarType(0),
259 w}; // barycentric coordinates (1-w,0,w)
260 }
261 // Check if P in edge region of BC, if so return projection of P onto BC
262 ScalarType const va = d3 * d6 - d5 * d4;
263 if (va <= ScalarType(0) and (d4 - d3) >= ScalarType(0) and (d5 - d6) >= ScalarType(0))
264 {
265 ScalarType const w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
266 return mini::SVector<ScalarType, 3>{
267 ScalarType(0),
268 ScalarType(1) - w,
269 w}; // barycentric coordinates (0,1-w,w)
270 }
271 // P inside face region. Compute Q through its barycentric coordinates (u,v,w)
272 ScalarType const denom = ScalarType(1) / (va + vb + vc);
273 ScalarType const v = vb * denom;
274 ScalarType const w = vc * denom;
275 return mini::SVector<ScalarType, 3>{
276 ScalarType(1) - v - w,
277 v,
278 w}; // = u*a + v*b + w*c, u = va * denom = ScalarType(1)0f-v-w
279}

283 mini::CMatrix TMatrixA,

285 mini::CMatrix TMatrixC>
286PBAT_HOST_DEVICE auto
287PointInTriangle(TMatrixP const& P, TMatrixA const& A, TMatrixB const& B, TMatrixC const& C)
288 -> mini::SVector<typename TMatrixP::ScalarType, TMatrixP::kRows>
289{
290 auto uvw = UvwPointInTriangle(P, A, B, C);
291 return A * uvw(0) + B * uvw(1) + C * uvw(2);
292}

294template <
295 mini::CMatrix TMatrixP,
296 mini::CMatrix TMatrixA,
297 mini::CMatrix TMatrixB,
298 mini::CMatrix TMatrixC,
299 mini::CMatrix TMatrixD>
300PBAT_HOST_DEVICE auto PointInTetrahedron(
301 TMatrixP const& P,
302 TMatrixA const& A,
303 TMatrixB const& B,
304 TMatrixC const& C,
305 TMatrixD const& D) -> mini::SVector<typename TMatrixP::ScalarType, TMatrixP::kRows>
306{
307 using ScalarType = typename TMatrixP::ScalarType;
308 auto constexpr kRows = TMatrixP::kRows;
309 auto constexpr kDims = 3;
310 static_assert(kRows == kDims, "This overlap test is specialized for 3D");
311
315
316 // Start out assuming point inside all halfspaces, so closest to itself
317 mini::SVector<ScalarType, kDims> X = P;
318 ScalarType d2min = std::numeric_limits<ScalarType>::max();
319
320 auto const PointOutsidePlane = [](auto const& p, auto const& a, auto const& b, auto const& c) {
321 ScalarType const d = Dot(p - a, Cross(b - a, c - a));
322 return d > ScalarType(0);
323 };
324 auto const TestFace = [&](auto const& a, auto const& b, auto const& c) {
325 // If point outside face abc then compute closest point on abc
326 if (PointOutsidePlane(P, a, b, c))
327 {
328 auto const Q = PointInTriangle(P, a, b, c);
329 ScalarType const d2 = SquaredNorm(Q - P);
330 // Update best closest point if (squared) distance is less than current best
331 if (d2 < d2min)
332 {
333 d2min = d2;
334 X = Q;
335 }
336 }
337 };
338 TestFace(A, B, D);
339 TestFace(B, C, D);
340 TestFace(C, A, D);
341 TestFace(A, C, B);
342 return X;
343}
