183template <class TDerived, class TBoundingVolume, class TPrimitive, int Dims>
184inline void BoundingVolumeHierarchy<TDerived, TBoundingVolume, TPrimitive, Dims>::Construct(
185 std::size_t nPrimitives,
186 std::size_t maxPointsInLeaf)
187{
188 Matrix<Dims, Eigen::Dynamic> P(Dims, nPrimitives);
189 for (auto p = 0; p < P.cols(); ++p)
190 {
191 P.col(p) = PrimitiveLocation(Primitive(p));
192 }
193 mKdTree.Construct(P, maxPointsInLeaf);
194 std::size_t const nBoundingVolumes = mKdTree.Nodes().size();
195 mBoundingVolumes.resize(nBoundingVolumes);
196 tbb::parallel_for(std::size_t{0ULL}, nBoundingVolumes, [this](std::size_t b) {
197 mBoundingVolumes[b] = BoundingVolumeOf(mKdTree.PointsInNode(static_cast<Index>(b)));
198 });
199}

220template <class TDerived, class TBoundingVolume, class TPrimitive, int Dims>
221template <class FIntersectsBoundingVolume, class FIntersectsPrimitive>
222inline std::vector<Index>
223BoundingVolumeHierarchy<TDerived, TBoundingVolume, TPrimitive, Dims>::PrimitivesIntersecting(
224 FIntersectsBoundingVolume&& ibv,
225 FIntersectsPrimitive&& ip,
226 std::size_t reserve) const
227{
228 std::vector<Index> intersectingPrimitives{};
229 intersectingPrimitives.reserve(reserve);
230
231 mKdTree.DepthFirstSearch([&](Index bvIdx, KdTreeNode const& node) -> bool {
232 if (node.IsLeafNode())
233 {
234 for (auto const idx : mKdTree.PointsInNode(node))
235 {
236 bool const bIntersects = ip(Primitive(idx));
237 if (bIntersects)
238 {
239 intersectingPrimitives.push_back(idx);
240 }
241 }
242 return false; // Cannot visit deeper than a leaf node
243 }
244 else
245 {
246 auto const bvIdxStl = static_cast<std::size_t>(bvIdx);
247 BoundingVolumeType const& bv = mBoundingVolumes[bvIdxStl];
248 return ibv(bv); // Visit deeper if this bounding volume overlaps with the
249 // queried shape
250 }
251 });
252
253 return intersectingPrimitives;
254}

257template <class FDistanceToBoundingVolume, class FDistanceToPrimitive>
258inline auto
259BoundingVolumeHierarchy<TDerived, TBoundingVolume, TPrimitive, Dims>::NearestPrimitivesTo(
260 FDistanceToBoundingVolume&& db,
261 FDistanceToPrimitive&& dp,
262 std::size_t K) const -> std::pair<std::vector<Index>, std::vector<Scalar>>
263{
264 std::vector<Index> neighbours{};
265 std::vector<Scalar> distances{};
266 neighbours.reserve(K);
267 distances.reserve(K);
268
269 enum class EQueueItem { Volume, Primitive };
270 struct QueueItem
271 {
272 EQueueItem type; // Indicates if this QueueItem holds a primitive or a volume
273 Index idx; // Index of the primitive, if this QueueItem holds a primitive, or index
274 // of the node, if this QueueItem holds a volume (recall that node_idx =
275 // bv_idx + 1)
276 Scalar d; // Distance from this QueueItem to p
277 };
278 auto const MakeVolumeQueueItem = [&](Index bvIdx) {
279 auto const bvIdxStl = static_cast<std::size_t>(bvIdx);
280 BoundingVolumeType const& bv = mBoundingVolumes[bvIdxStl];
281 Scalar const d = db(bv);
282 QueueItem const q{EQueueItem::Volume, bvIdx, d};
283 return q;
284 };
285 auto const MakePrimitiveQueueItem = [&](Index pIdx) {
286 PrimitiveType const& p = Primitive(pIdx);
287 Scalar const d = dp(p);
288 QueueItem const q{EQueueItem::Primitive, pIdx, d};
289 return q;
290 };
291
292 auto const Greater = [](QueueItem const& q1, QueueItem const& q2) {
293 return q1.d > q2.d;
294 };
295 using PriorityQueue = std::priority_queue<QueueItem, std::vector<QueueItem>, decltype(Greater)>;
296 PriorityQueue queue{Greater};
297 queue.push(MakeVolumeQueueItem(0));
298 auto const& nodes = mKdTree.Nodes();
299 while (!queue.empty())
300 {
301 QueueItem const q = queue.top();
302 queue.pop();
303 if (q.type == EQueueItem::Volume)
304 {
305 auto const qIdxStl = static_cast<std::size_t>(q.idx);
306 KdTreeNode const& node = nodes[qIdxStl];
307 if (node.IsLeafNode())
308 {
309 for (auto const pIdx : mKdTree.PointsInNode(node))
310 {
311 queue.push(MakePrimitiveQueueItem(pIdx));
312 }
313 }
314 else
315 {
316 if (node.HasLeftChild())
317 queue.push(MakeVolumeQueueItem(node.lc));
318 if (node.HasRightChild())
319 queue.push(MakeVolumeQueueItem(node.rc));
320 }
321 }
322 else
323 {
324 // If the queue item at the beginning of the priority queue is a primitive,
325 // it means that primitive is closer to the point p than any other primitive or
326 // bounding volume in the queue, thus it is the current closest primitive to p.
327 neighbours.push_back(q.idx);
328 distances.push_back(q.d);
329 }
330
331 if (neighbours.size() == K)
332 break;
333 }
334 return {neighbours, distances};
335}

338template <
339 class TDerived2,

346inline IndexMatrixX
347BoundingVolumeHierarchy<TDerived, TBoundingVolume, TPrimitive, Dims>::OverlappingPrimitivesImpl(
348 BoundingVolumeHierarchy<TDerived2, TBoundingVolume2, TPrimitive2, Dims2> const& other,
349 FBoundingVolumesOverlap&& bvo,
350 FPrimitivesOverlap&& po,
351 FPrimitivesAreAdjacent&& pa,
352 std::size_t reserve) const
353{
354 using BoundingVolumeType2 = TBoundingVolume2;
355 using PrimitiveType2 = TPrimitive2;
356 std::vector<Index> overlaps{};
357 overlaps.reserve(reserve * 2);
358 auto const& nodes1 = mKdTree.Nodes();
359 auto const& nodes2 = other.mKdTree.Nodes();
360
361 using PrimitivePairType = std::pair<Index, Index>;
362 std::stack<PrimitivePairType> stack{};
363 stack.push({0, 0}); // Root bounding volumes of *this and other
364 while (!stack.empty())
365 {
366 auto const [n1, n2] = stack.top();
367 auto const n1Stl = static_cast<std::size_t>(n1);
368 auto const n2Stl = static_cast<std::size_t>(n2);
369 stack.pop();
370 BoundingVolumeType const& bv1 = mBoundingVolumes[n1Stl];
371 BoundingVolumeType2 const& bv2 = other.mBoundingVolumes[n2Stl];
372 if (!bvo(bv1, bv2))
373 continue;
374
375 KdTreeNode const& node1 = nodes1[n1Stl];
376 KdTreeNode const& node2 = nodes2[n2Stl];
377 bool const bIsNode1Leaf = node1.IsLeafNode();
378 bool const bIsNode2Leaf = node2.IsLeafNode();
379 if (bIsNode1Leaf and bIsNode2Leaf)
380 {
381 for (auto const p1 : mKdTree.PointsInNode(node1))
382 {
383 for (auto const p2 : other.mKdTree.PointsInNode(node2))
384 {
385 // For self collision, prevent expensive narrow phase testing when mesh
386 // primitives are adjacent, since this gives false positives
387 if (pa(Primitive(p1), other.Primitive(p2)))
388 continue;
389
390 if (po(Primitive(p1), other.Primitive(p2)))
391 {
392 overlaps.push_back(p1);
393 overlaps.push_back(p2);
394 }
395 }
396 }
397 }
398 else if (bIsNode1Leaf and not bIsNode2Leaf)
399 {
400 if (node2.HasLeftChild())
401 stack.push({n1, node2.lc});
402 if (node2.HasRightChild())
403 stack.push({n1, node2.rc});
404 }
405 else if (not bIsNode1Leaf and bIsNode2Leaf)
406 {
407 if (node1.HasLeftChild())
408 stack.push({node1.lc, n2});
409 if (node1.HasRightChild())
410 stack.push({node1.rc, n2});
411 }
412 else
413 {
414 if (node1.depth < node2.depth)
415 {
416 if (node1.HasLeftChild())
417 stack.push({node1.lc, n2});
418 if (node1.HasRightChild())
419 stack.push({node1.rc, n2});
420 }
421 else
422 {
423 if (node2.HasLeftChild())
424 stack.push({n1, node2.lc});
425 if (node2.HasRightChild())
426 stack.push({n1, node2.rc});
427 }
428 }
429 }
430 return common::ToEigen(overlaps).reshaped(2, overlaps.size() / 2);
431}
