33template <class T>
34concept CLinearOperator = requires(T t)
35{
36 {
37 t.OutputDimensions()
38 } -> std::convertible_to<int>;
39 {
40 t.InputDimensions()
41 } -> std::convertible_to<int>;
42 {t.Apply(VectorX{}, std::declval<VectorX&>())};
43 {t.Apply(MatrixX{}, std::declval<MatrixX&>())};
44 {
45 t.ToMatrix()
46 } -> std::convertible_to<CSCMatrix>;
47};

197template <CLinearOperator... TLinearOperators>
198inline LinearOperator<TLinearOperators...>::LinearOperator(TLinearOperators const&... inOps)
199 : ops(std::make_tuple(std::cref(inOps)...))
200{
201 bool const bInputDimensionsMatch = std::apply(
202 [this](auto... op) -> bool {
203 return ((InputDimensions() == op.InputDimensions()) and ...);
204 },
205 ops);
206 bool const bOutputDimensionsMatch = std::apply(
207 [this](auto... op) -> bool {
208 return ((OutputDimensions() == op.OutputDimensions()) and ...);
209 },
210 ops);
211 if (not(bInputDimensionsMatch and bOutputDimensionsMatch))
212 {
213 throw std::invalid_argument(
214 "Dimensionality mismatch found in CompositeLinearOperator's linear operators.");
215 }
216}
